# Firebase Configuration
# Get these values from your Firebase Console: https://console.firebase.google.com/
# 1. Create a new Firebase project
# 2. Go to Project Settings > General
# 3. Scroll down to "Your apps" and click "Web app" (</>)
# 4. Copy the config values below

NEXT_PUBLIC_FIREBASE_API_KEY=your-api-key-here
NEXT_PUBLIC_FIREBASE_AUTH_DOMAIN=your-project-id.firebaseapp.com
NEXT_PUBLIC_FIREBASE_PROJECT_ID=your-project-id
NEXT_PUBLIC_FIREBASE_STORAGE_BUCKET=your-project-id.appspot.com
NEXT_PUBLIC_FIREBASE_MESSAGING_SENDER_ID=your-sender-id
NEXT_PUBLIC_FIREBASE_APP_ID=your-app-id

# IMPORTANT: After adding your Firebase config:
# 1. Enable Google Sign-In in Firebase Console > Authentication > Sign-in method
# 2. Enable Phone Authentication (optional)
# 3. Enable Email Link Authentication (optional)
# 4. Copy this file to .env.local and fill in your actual values
